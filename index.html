/** * Valida si un número dado es primo. * @param {number} numero - El número a evaluar. * @returns {boolean} - True si es primo, false en caso contrario. */ function esPrimo(numero) { console.log(`--- Validando si ${numero} es primo ---`); // Casos base: menores o iguales a 1 no son primos if (numero <= 1) { console.log(`Depuración: ${numero} es menor o igual a 1, no es primo.`); return false; } // Iteración optimizada hasta la raíz cuadrada for (let i = 2; i <= Math.sqrt(numero); i++) { if (numero % i === 0) { console.log(`Depuración: Divisible entre ${i}. No es primo.`); return false; } } console.log(`Depuración: No se encontraron divisores para ${numero}. Es primo.`); return true; } // Ejecución de prueba let numeroAProbar = 29; let resultadoPrimo = esPrimo(numeroAProbar); console.log(`¿El número ${numeroAProbar} es primo?:`, resultadoPrimo);
